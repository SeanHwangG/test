> Question

* Print location of cup at the end of simulation
* A reverse 1, 2 / B reverse 2, 3 / C reverse 1, 3

```txt
Input: AB
Output: 3

Input: CBABCACCC
Output: 1
```

{% endtab %}
{% tab title='BJ_3028.py' %}

```py
cur = 1
for move in input():
  if move == 'A':
    if cur == 1:
      cur = 2
    elif cur == 2:
      cur = 1
  elif move == 'B':
    if cur == 2:
      cur = 3
    elif cur == 3:
      cur = 2
  else:
    if cur == 1:
      cur = 3
    elif cur == 3:
      cur = 1
print(cur)
```

{% tab title='BJ_3028.cpp' %}

```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
  int a, b, c;
  a = 1, b = c = 0;
  string s;
  cin >> s;

  for (auto i : s) {
    if (i == 'A') swap(a, b);
    if (i == 'B') swap(b, c);
    if (i == 'C') swap(a, c);
  }
  if (a) cout << 1 << '\n';
  if (b) cout << 2 << '\n';
  if (c) cout << 3 << '\n';
}
```
