#include <iostream>
#include <string>
using namespace std;

int c;
void add(string& a, string& b) {
  int i = a.size(), j = b.size();
  while (--j >= 0) {
    a[--i] += b[j] + c - '0';
    a[i] > '9' ? a[i] -= 10, c = 1 : c = 0;
  }
  while (--i >= 0 && c) {
    a[i] += c;
    a[i] > '9' ? a[i] -= 10, c = 1 : c = 0;
  }
  if (c)
    a.insert(a.begin(), '1');
  cout << a;
}
int main(){
  int i, j;
  string a, b;
  cin >> a >> b;
  a.size()>b.size() ? add(a, b) : add(b, a);
}
