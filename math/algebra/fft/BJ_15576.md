#include<bits/stdc++.h>
using namespace std;

const double PI = acos(-1);
using cpx = complex<double>;
using poly = vector<cpx>;

void fft(poly& pol, cpx w){
  int n=pol.size();
  if (n==1) return;

  int i;

  poly e(n >> 1), o(n >> 1);
  for(i=0;i<n;++i){
    ((i & 1)? o: e)[i>>1]=pol[i];
  }

  fft(e, w*w);
  fft(o ,w*w);

  cpx p(1,0);
  for(i=0; i<(n>>1); ++i){
    pol[i] = e[i] + p * o[i];
    pol[i+(n>>1)] = e[i]-p*o[i];
    p *= w;
  }
}

poly mult(poly a,poly b){
  int n=1;
  for(;n<a.size()+1||n<b.size()+1; n<<=1);
  n<<=1;
  a.resize(n);
  b.resize(n);
  poly c(n);

  cpx w(cos(2*PI/n),sin(2*PI/n));

  fft(a,w);
  fft(b,w);

  int i=0;

  for(i=0;i<n;++i){
    c[i]=a[i]*b[i];
  }

  fft(c,cpx(1,0)/w);
  for(i=0;i<n;++i){
    c[i]/=cpx(n,0);
    c[i]=cpx(round(c[i].real()),round(c[i].imag()));
  }

  return c;
}

poly StoP(string s){
  poly r;
  for(int i= s.size() - 1;i >= 0; --i)
    r.push_back(cpx(s[i]-'0',0));
  return r;
}

int main(void){
  string a,b;
  cin >> a >> b;

  if(a == "0" || b == "0"){
    cout << 0;
    return 0;
  }

  poly r = mult(StoP(a),StoP(b));
  stack<int> st;
  int up = 0;
  for (cpx i: r){
    int rp = (int) i.real();
    st.push((rp % 10 + up) % 10);
    up = (rp + up) / 10;
  }

  int z = 1;
  while(!st.empty()){
    if(z && st.top()) z=0;
    if(!z)  cout<<st.top();
    st.pop();
  }
}
