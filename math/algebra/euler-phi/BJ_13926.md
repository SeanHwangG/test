> Question

* print number of integers that stisfies gcd(n, k) = 1 (1 ≤ k ≤ n)

```txt
Input: 10
Output: 4
```

{% endtab %}

```py
from math import gcd
a = int(input())

def f(x, mod):
  return (x * x + 1) % mod

def getfac(su):
  for i in range(2, 10**5):
    if su % i == 0:
      return i
    if i * i > su:
      return su

  p, q = 2, 2
  for i in range(1, 11**5):
    p = (p ** 2 + 1) % su
    q = (((q ** 2 + 1) % su) ** 2 + 1) % su
    d = gcd(p - q, su)
    if 1 < d < su:
      return d
  return su

r = 1
while a > 1:
  p = getfac(a)
  if p < a:
    if p > 10**9:
      p = a // p
    p = getfac(p)

  x = 1
  while a % p == 0:
    r *= p - x
    a //= p
    x = 0

print(r)
```
