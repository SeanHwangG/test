> Question

* Find number between A, B which is coprime with N

```txt
Input:
2
1 10 2
3 15 5

Output:
Case #1: 5
Case #2: 10
```

> Solution

* Remove all multiples of that are prime of N

{% endtab %}
{% tab title='BJ_9359.cpp' %}

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main() {
  int tt; cin >> tt;
  for (int tc = 1; tc <= tt; tc++) {
    ll a, b, n;
    cin >> a >> b >> n;
    ll ans = b - a + 1;

    ll temp = n;
    vector<ll> comp;
    for (ll i = 2; i * i <= n; i++) {
      if (temp % i == 0) {
        comp.push_back(i);
        while (temp % i == 0) temp /= i;
      }
    }
    if (temp != 1) comp.push_back(temp);

    ll k = comp.size(), sum = 0;
    for (ll i = 1; i < (1ll << k); i++) {
      ll cnt = 0, lcm = 0;
      for (ll j = 0; j < k; j++) {
        if (i & (1 << j)) {
          cnt++;
          if (lcm == 0) lcm = comp[j];
          else lcm = lcm * comp[j] / __gcd(lcm, comp[j]);
        }
      }
      ll d = b / lcm - (a - 1) / lcm;
      if (cnt % 2 == 1) sum += d;
      else if (cnt > 0) sum -= d;
    }
    cout << "Case #" << tc << ": " << ans - sum << '\n';
  }
}
```
