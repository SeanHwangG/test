> Question

* a permutation consisting of numbers from 1 to N, find the permutations that follow in alphabetical order

```txt
Input:
4
1 2 3 4
Output: 1 2 4 3
```

{% endtab %}
{% tab title='BJ_10972.cpp' %}

```cpp
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
  int num, a;
  cin >> num;
  vector<int> v;
  for (int i = 0; i < num; i++) {
    cin >> a;
    v.push_back(a);
  }

  if (next_permutation(v.begin(), v.end()))
    for (int i = 0; i < v.size(); i++)
      cout << v[i] << " ";
  else
    cout << -1;

  return 0;
}
```

```py
N = int(input())
A = list(map(int, input().split()))

i = N-1
while i > 0 and A[i - 1] > A[i]:
  i -= 1

if i == 0:
  print(-1)
  exit()

j = N-1
while A[i-1]>A[j]:
  j -= 1

A[i-1], A[j] = A[j], A[i-1]
print(*(A[:i]+list(reversed(A[i:]))))
```
