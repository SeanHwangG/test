> Question

* Perform binary search on sorted array if duplicate is possible

```txt
Input: nums = [2,2,2,0,1]
Output: 0
```

{% endtab %}
{% tab title='LC_154.cpp' %}

```cpp
// O(N): Worst case
int findMin(vector<int> &num) {
  int lo = 0, hi = num.size() - 1, mid = 0;

  while (lo < hi) {
    if (num[mid] > num[hi])
      lo = mid + 1;
    else if (num[mid] < num[hi])
      hi = mid;
    else
      hi--;
  }

  return num[lo];
}
```
